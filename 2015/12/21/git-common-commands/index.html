git常用命令

#配置git
$ git config --global user.name "K-John"
$ git config --global user.email "19941222hb@gmail.com"
#初始化 git workspace(就是你当前所在的目录)
$ git init
#将当前工作区的文件加到暂存区,实际上是将文件的改动加到暂存区
$ git add README
$ #将暂存区的内容添加到当前分支所对应的分支的版本库中(git的仓库)
$ git commit -m "desriptions"
#查看仓库当前的状态
$ git status
#修改某一文件,但是没有添加到缓冲区,此时可以用git diff 查看修改前和未修改前的区别
$ git diff README
$ git diff HEAD -- README #查看版本库中最新版本文件与工作区文件的区别
#查看每一次commit的详情
$ git log [--pretty=oneline]
#回退到上一次commit
$ git reset  --hard HEAD^
#回退到上上一次commit(没回退一次相应的commit详情都会自动消失)
#HEAD 相当于一个指针,总是指向当前的分支的指针(分支名代表一个分支的指针),而当前分支指针指向commit,(HEAD-->分支指针--->commit) 改变HEAD就相当于改变>    分支指针 所指向的commit,这样就可以在不同commit之间切换,每次提交分支指针都会向前移动一步
$ git reset --hard HEAD^^
#查看当前每一次对commit的操作,用于回到未来某一版本时查看commit id
$ git reflog
#回退到历史版本,又想要回到未来某一版本
$ git  reset --hard 3628164 #(这一串数字为commit id的前几位数字,这样就不用将整个commit id写出来,通过git reflog查看未来的commit id)
#撤销工作区内对文件的修改(前提是在你没有将文件添加到暂存区)
$ git checkout -- README
#撤销暂存区的修改,并重新将修改放回工作区
$ git reset HEAD  README 
#从版本库中删除某一文件
$ git rm LICENSE
$ git commit -m "deleted LICENSE"
#从版本库中恢复已经删除的文件,只需要回到以前的版本
#创建ssh key
$ ssh-keygen -t rsa -C "19941222hb@gmail.com"
#将本地仓库关联到github的远程仓库(前提是已经将你电脑的ssh key添加到github的SSH key列表)
$ git remote add origin git@github.com:free-free/algorithm.git   #origin 为远程仓库的叫法(只是为了方面操作),也可以自己取名;free-free为github账号名;a    lgorithm为你想要关 联的仓库,必须现在现在github上创建
#将本地版本库推送到远程github库
$ git push -u origin master # -u 参数 Git不但会把本地的master分支内容推送的远程新的master分支，还会把本地的master分支和远程的master分支关联起来 
#推送非主分支到到远程库,github会自动建立一个dev分支与你本地的dev分支对应
$ git push origin dev
#将远程的仓库克隆到本地
$ git clone git@github.com:free/algorithm.git
#git创建一个分支只需要重新创建一个分支指针,并且将该分支指针指向创建时所对应的commit,分支间切换就只需要让HEAD指向另一分支指针,分支切换后对文件的修>    改就指向该分支了
#创建分支,并且切换到分支
$ git checkout -b dev #相当于 git branch dev git checkout dev
#切换到指定分支
$ git checkout dev
#查看分支数
$ git branch
#创建新分支 
$ git branch dev
#删除分支
$ git branch -d dev
#强制删除一个没有合并的分支
$ git branch -D dev
#合并指定分支到当前分支(假设当前出master分支,则下面指令将dev分支合并到master分支)
$ git merge dev #需要自己解决合并冲突
#合并分支时，如果可能，Git会用Fast forward模式，但这种模式下，删除分支后，会丢掉分支信息。如果要强制禁用Fast forward模式，Git就会在merge时生成一个新的commit，这样，从分支历史上就可以看出分支信息。--no-ff(禁用Fast forward mode)方式的git merge
$ git merge --no-ff -m "merge dev into master"  dev
#假设你在dev分支上工作,突然需要跳到bug分支去修复bug,但是你当前的工作区的工作还没#完成,但是你又不想要提交修改,这是你可以通过以前命令将当前工作保存>    起来,然后切换到其他bug分支去修复bug,
$ git stash
#当你把bug修复完之后,切换到刚才工作的分支,这是可以通过命令去恢复刚才没有提交的工作区
$ git stash list #查看所有的stash
$ git stash apply # 恢复刚才未提交的工作, 但是没有将stash删除,这是需要通过以下命令将stash 的内容删除
$ git stash drop
$ git stash pop # 恢复stash的内容,并且删除stash的内容,相当于git stash apply,git stash drop
# 查看远程库的信息
$ git remote
$ git remote -v #更详细
# 新建分支并将远程仓库某一个分支克隆到当前分支
$ git checkout -b dev origin/dev
# 当远程仓库最新的提交与试图推送的提交有矛盾的时候,用git pull把最新的提交从origin/dev抓下来，然后，在本地合并，解决冲突，commit 之后,再推送
$ git pull
# 如果git pull提示“no tracking information”，则说明本地分支和远程分支的链接关系没有创建，用以下命令解决
$ git branch --set-upstream branch-name orgin/branch-name
# 为当前的版本打一个标签(标签其实也是一个指向commit指针,但是它不能改变),默认是为最新的commit打标签
$ git tag v1.0
$ git tag v1.1  6224937(commit id) #为某一次commit打标签
$ git tag -a v1.2 -m 'version 1.2 released' 3628164 #创建带说明的标签
# 查看所有标签(标签不是按时间顺序列出而是按字母顺序列出)
$ git tag
# 显示某一标签对应的详细信息
$ git show v1.0
# 删除某一标签
$ git tag -d v1.0
# 推送某个标签到远程仓库
$ git push origin v1.0
# 一次性推送全部尚未推送远程的本地标签
$ git push origin --tags
# 删除远程的标签
$ git push origin :refs/tags/v0.9
